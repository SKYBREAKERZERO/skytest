    strategy = "Rolling"

    preferences {
      min_healthy_percentage = 50
    }
  }

  tag {
    key                 = "Name"
    value               = "${var.env}-web"
    propagate_at_launch = true
  }

#!/bin/bash
set -euo pipefail

APP_NAME=${APP_NAME:-webapp}
REGION=${REGION:-ap-northeast-1}
IMAGE=${IMAGE:-nginx:latest}

echo "[INFO] Starting setup..."

# =========================
# 安装依赖
# =========================
yum update -y
yum install -y docker amazon-cloudwatch-agent aws-cli curl

systemctl enable docker
systemctl start docker

# =========================
# CloudWatch Logs group（防失败）
# =========================
aws logs create-log-group \
  --log-group-name "${APP_NAME}-app" \
  --region "${REGION}" || true

# =========================
# CloudWatch Agent 配置
# =========================
cat <<EOF > /opt/aws/amazon-cloudwatch-agent/etc/agent.json
{
  "metrics": {
    "namespace": "EC2/Custom",
    "append_dimensions": {
      "InstanceId": "\${aws:InstanceId}"
    },
    "metrics_collected": {
      "mem": {
        "measurement": ["mem_used_percent"]
      },
      "disk": {
        "measurement": ["used_percent"],
        "resources": ["/"]
      }
    }
  },
  "logs": {
    "logs_collected": {
      "files": {
        "collect_list": [
          {
            "file_path": "/var/log/messages",
            "log_group_name": "${APP_NAME}-system",
            "log_stream_name": "{instance_id}"
          }
        ]
      }
    }
  }
}
EOF

# 启动 agent
/opt/aws/amazon-cloudwatch-agent/bin/amazon-cloudwatch-agent-ctl \
  -a fetch-config -m ec2 \
  -c file:/opt/aws/amazon-cloudwatch-agent/etc/agent.json -s

systemctl enable amazon-cloudwatch-agent
docker pull ${IMAGE}

# =========================
# systemd 服务
# =========================
cat <<EOF > /etc/systemd/system/${APP_NAME}.service
[Unit]
Description=${APP_NAME}
After=docker.service
Requires=docker.service

[Service]
Restart=always
ExecStart=/usr/bin/docker run \
  --name ${APP_NAME} \
  --restart always \
  -p 80:80 \
  --log-driver=awslogs \
  --log-opt awslogs-region=${REGION} \
  --log-opt awslogs-group=${APP_NAME}-app \
  ${IMAGE}

ExecStop=/usr/bin/docker stop ${APP_NAME}
ExecStopPost=/usr/bin/docker rm ${APP_NAME}

[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl enable ${APP_NAME}
systemctl start ${APP_NAME}

# =========================
# 健康检查
# =========================
for i in {1..10}; do
  if curl -sf http://localhost/ > /dev/null; then
    echo "[INFO] App healthy"
    exit 0
  fi
  sleep 5
done

echo "[ERROR] App failed"
exit 1
